# 문제
>숫자 카드 2

https://www.acmicpc.net/problem/10816

# 코드
```java
import java.util.*;
import java.util.stream.Collectors;

public class Test {
    public static void main(String[] args) {
        List<Integer> cards = Arrays.asList(6, 3, 2, 10, 10, 10, -10, -10, 7, 3);
        List<Integer> numbers = Arrays.asList(10, 9, -5, 2, 3, 4, 5, -10);

        Map<Integer, Integer> numberMap = numbers.stream()
                .collect(Collectors.toMap(
                        number -> number,
                        number -> cards.stream().mapToInt(card -> card == number ? 1 : 0).sum()
                ));
    }
}
```
